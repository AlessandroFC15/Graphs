9ª Questão

A questão nº 9 da lista pedia a elaboração de um programa computacional que informasse o caminho mais curto entre 2 vértices dado o dígrafo da questão. O comando pedia o uso de busca em largura para encontrar o resultado, além de apresentar um log de execução.

Comando da Questão


Dado o grafo não direcionado G abaixo:



Dado o dígrafo D como entrada, elabore um programa computacional que informe o caminho mais curto entre dois dos seus vértices. Use busca em largura e apresente um log com exemplo de execução.

Procedimento

GrafoListaAdjacencia.prototype.caminhoMaisCurto = function(origem, destino) {
    if (this.existeVertice(origem) && this.existeVertice(destino)) {
        var buscaEmLargura = this.buscaEmLargura(origem);

        if (origem == destino) {
            return [origem];
        } else if (! buscaEmLargura[destino] || buscaEmLargura[destino].antecessor == -1) {
            return null;
        } else {
            return this.caminhoMaisCurto(origem, buscaEmLargura[destino].antecessor).concat(destino);
        }
    } else {
        return null;
    }
};
Primeiramente, verificamos se os vérticos inseridos como origem e destino realmente existem no grafo. Em seguida, realizamos a busca em largura partindo da origem e temos 3 casos possíveis. Caso a origem seja igual ao destino, retornamos o próprio vértice. Caso a busca em largura não possua o vértice destino ou a busca não alcançou o vértice, retornamos null. Por último, retornamos o caminho mais curto entre origem e o antecessor do destino, concatenado com o destino. 
GrafoListaAdjacencia.prototype.buscaEmLargura = function(verticeEscolhido) {
    if (this.existeVertice(verticeEscolhido)) {
        var grafoBusca = this.iniciarBuscaEmLargura(verticeEscolhido);

        var Q = [grafoBusca[verticeEscolhido]];

        while (Q.length > 0) {
            var u = Q[0];

            Q = Q.slice(1);

            var adj = this.getVerticesAdjacentes(u.indice);

            for (var i in adj) {
                var v = grafoBusca[adj[i]];

                if (v.cor == 'BRANCO') {
                    v.cor = 'CINZA';
                    v.distancia = u.distancia + 1;
                    v.antecessor = u.indice;
                    Q.push(v);
                }
            }

            u.cor = 'PRETO';
        }

        return grafoBusca;
    } else {
        console.log("Vértice inválido");
        return null;
    }
};
GrafoListaAdjacencia.prototype.iniciarBuscaEmLargura = function(verticeEscolhido) {
    var grafoBusca = [];

    for (var i = 0; i < this.numVertices; i++) {
        var vertice = {
            cor: 'BRANCO',
            distancia: Infinity,
            antecessor: -1,
            indice: i
        };

        grafoBusca.push(vertice);
    }

    grafoBusca[verticeEscolhido].cor = 'CINZA';
    grafoBusca[verticeEscolhido].distancia = 0;

    return grafoBusca;
};

O método iniciarBuscaEmLargura initializa todos os vértices do grafo com cor branca, distância infinita e antecessor com valor -1. Em seguida, o vértico escolhido para iniciar a busca tem sua cor setada para cinza e distância 0.

O método principal da busca em largura chama o método para iniciar a busca e em seguida, enquanto a lista Q não for vazia, a busca pega o primeiro elemento, o retira de Q e visita todos os seus vértices adjacentes, caso ainda não tenham sido visitados.
